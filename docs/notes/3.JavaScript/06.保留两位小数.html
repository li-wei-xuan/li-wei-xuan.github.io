# toFix(n)

//四舍五入

console.log(num.toFixed(2)) //123.12

console.log(num.toFixed(2)) //123.13

# parseInt()

var num = 123.123
console.log(parseInt(num * 100) / 100)  //123.12
//尾数为零不保留
var num = 123.003
console.log(parseInt(num * 100) / 100)  //123
//尾数为零保留

console.log((Math.floor(num * 100) / 100).toFixed(2))  //123.00
